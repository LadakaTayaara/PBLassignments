%macro% rw 4
    mov rax, %1
    mov rdi, %2
    mov rsi, %3
    mov rdx, %4
    syscall
%endmacro%

section .data
    msg db "the decimal sum is",0ah
    msglen equ $-msg

    arr db 40,50,60,70,80
    arrlen equ $-arr

    result db "00000",0ah
    reslen equ $-reslen

section .text
    global _start

_start:
    rw 1,1,msg,msglen
    mov ax,0
    mov rsi,arr
    mov cl,arrlen

nextno:
    add al,[rsi]
    jnc nocarry
    inc ah

nocarry:
    inc rsi
    dec cl
    jnz nextno

    mov [rsi],result+4
    mov bx,10

convert:
    mov dx,0
    div bx
    add dl,30h
    mov [rsi],dl
    dec rsi
    cmp ax,0
    jnz nextno

    rw 1, 1, result_ascii, result_ascii_len

    mov rax, 60 
    mov rdi, 0 
    syscall
